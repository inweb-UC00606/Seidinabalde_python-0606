palavra = "python"
letras = list(palavra)          # array com as letras da palavra
mostrar = ["_"] * len(letras)   # array para mostrar o progresso
erros = 0

print("JOGO DA PALAVRA SECRETA")

while erros < 5 and "_" in mostrar:
    print("Palavra:", " ".join(mostrar))
    letra = input("Letra: ").lower()

    if letra in letras:
        print("Certo!")
        for i in range(len(letras)):
            if letras[i] == letra:
                mostrar[i] = letra
    else:
        erros += 1
        print("Errado! Erros:", erros)

if "_" not in mostrar:
    print("Ganhou! A palavra era:", palavra)
else:
    print("Perdeu! A palavra era:", palavra)
